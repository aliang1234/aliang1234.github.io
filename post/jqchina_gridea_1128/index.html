JAVASCRIPT

在JS代码中，ripple()函数用于制作登录按钮的点击波特效。

function ripple(elem, e) {
$('.ripple').remove();
var elTop = elem.offset().top, elLeft = elem.offset().left, x = e.pageX - elLeft, y = e.pageY - elTop;
var $ripple = $('<div class=\'ripple\'></div>');
$ripple.css({
top: y,
left: x
});
elem.append($ripple);
}
