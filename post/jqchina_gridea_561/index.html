使用方法

初始化插件

在页面DOM元素加载完毕之后，通过下面的方法来使鼠标滑过相应的省份时将其高亮。

$(document).ready(function(){
maphover();
});

function maphover(){

var self = "";

$(".city").hover(

function(){

self = $(this);

self.addClass("hover").children("div").show();

},

function(){

self = (this);
self.children("div").hide();
self.removeClass("hover");
}
);
(".city").click(

function(){

self = $(this);

self.addClass("hover").children("div").show();

},

function(){

self = $(this);

self.children("div").hide();

self.removeClass("hover");

}

);


};
