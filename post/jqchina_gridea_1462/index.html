首先定义Accordion对象：

var Accordion = function(el, multiple) {
this.el = el || {};
this.multiple = multiple || false;

// Variables privadas

var links = this.el.find('.link');

// Evento

links.on('click', {el: this.el, multiple: this.multiple}, this.dropdown)

}
