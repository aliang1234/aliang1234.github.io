初始化插件

在页面DOM元素加载完毕之后，可以通过下面的方法来初始化该jquey轮播图插件。

$('#pbSlider0').pbTouchSlider({
slider\_Wrap : '#yourId', // Assign a unique ID to the div.o-sliderContainer
slider\_Item : '.o-slider--item', // Single Item
slider\_Drag : true, // Your choise.. to dragIt or not to dragIt..this is the question...
slider\_Dots : { // Wanna see dots or not?
class :'.o-slider-pagination',
enabled : true
},
slider\_Arrows : { // Wanna see Arrows or not?
class :'.o-slider-arrows',
enabled : true
},
slider\_Threshold : 25, // Percentage ofdragX before go to next/prev slider
slider\_Speed : 1000,
slider\_Ease : 'cubic-bezier(0.5, 0, 0.5, 1)',// see http://cubic-bezier.com/
slider\_Breakpoints : { // Kind of media queries ( can add more if you know how to do it :D and if you need )
default : {
height : 500 //height on desktop
},
tablet : {
height : 400, // height on tablet
media : 1024 // tablet breakpoint
},
smartphone : {
height : 300, // height on smartphone
media : 768 // smartphone breakpoint
}
}
});
